74 class FDGenericBuffer {

89  FDGenericBuffer(size_t size, nvinfer1::DataType type)

103 
104  FDGenericBuffer(FDGenericBuffer&& buf)
105  : mSize(buf.mSize),
106  mCapacity(buf.mCapacity),
107  mType(buf.mType),
108  mBuffer(buf.mBuffer) {
109  buf.mSize = 0;
110  buf.mCapacity = 0;
111  buf.mType = nvinfer1::DataType::kFLOAT;
112  buf.mBuffer = nullptr;
113  }
114 
115  FDGenericBuffer& operator=(FDGenericBuffer&& buf) {
116  if (this != &buf) {
117  freeFn(mBuffer);
118  mSize = buf.mSize;
119  mCapacity = buf.mCapacity;
120  mType = buf.mType;
121  mBuffer = buf.mBuffer;
122  // Reset buf.
123  buf.mSize = 0;
124  buf.mCapacity = 0;
125  buf.mBuffer = nullptr;
126  }
127  return *this;
128  }

133  void* data() {
134  if (mExternal_buffer != nullptr) return mExternal_buffer;
135  return mBuffer;
136  }

142  if (mExternal_buffer != nullptr) return mExternal_buffer;
143  return mBuffer;
144  }

154  size_t nbBytes() const { return this->size() * TrtDataTypeSize(mType); }
155 
159  void SetExternalData(size_t size, nvinfer1::DataType type, void* buffer) {

168  void SetExternalData(const nvinfer1::Dims& dims, const void* buffer) {

172 
177  void resize(size_t newSize) {
178  mExternal_buffer = nullptr;
179  mSize = newSize;
180  if (mCapacity < newSize) {
181  freeFn(mBuffer);
182  if (!allocFn(&mBuffer, this->nbBytes())) {
183  throw std::bad_alloc{};
184  }
185  mCapacity = newSize;
186  }
187  }
188 
192  void resize(const nvinfer1::Dims& dims) { return this->resize(Volume(dims)); }
193 
194  ~FDGenericBuffer() {
195  mExternal_buffer = nullptr;
196  freeFn(mBuffer);
197  }

199  private:
200  size_t mSize{0}, mCapacity{0};
201  nvinfer1::DataType mType;
202  void* mBuffer;
203  void* mExternal_buffer;
204  AllocFunc allocFn;
205  FreeFunc freeFn;
206 };

238 struct ShapeRangeInfo {
239  explicit ShapeRangeInfo(const std::vector<int64_t>& new_shape) {
240  shape.assign(new_shape.begin(), new_shape.end());
241  min.resize(new_shape.size());
242  max.resize(new_shape.size());
243  is_static.resize(new_shape.size());
244  for (size_t i = 0; i < new_shape.size(); ++i) {
245  if (new_shape[i] > 0) {
246  min[i] = new_shape[i];
247  max[i] = new_shape[i];
248  is_static[i] = 1;
249  } else {
250  min[i] = -1;
251  max[i] = -1;
252  is_static[i] = 0;
253  }
254  }
255  }
